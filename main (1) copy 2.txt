#include <stdio.h>
#include <stdlib.h>

int main()
{
    int F, D, E, i;
    scanf("%d", &F);
    scanf("%d", &D);

    for(i = 0; i < D && F >= 0; i++)
    {
        scanf("%d", &E);
        F-=E;
        if(F < 0)
            i--;
    }
    if(F>=0)
        printf("Happy Cat!");
    else
        printf("%d", i);
    return 0;
}
