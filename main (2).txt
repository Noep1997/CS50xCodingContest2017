#include <stdio.h>
#include <stdlib.h>

int checkPlaces(long ** hotel, int requiered, int N, int M)
{
    int i, j, cpt = 0;
    for(i = 0; i < N && cpt < requiered; i++)
    {
        j = 0;
        cpt = 0;
        for(j = 0; j < M && cpt < requiered; j++)
            cpt = ((*(hotel[i]-(j/(sizeof(long)*8)))&(1L<<(j%(sizeof(long)*8)))) ? 0 : cpt +1);
    }
    return requiered == cpt;
}

int main()
{
    int M, N, i, requiered, j;
    scanf("%d", &N);
    long* hotel[N];
    scanf("%d", &M);
    char line[M];
    for(i = 0; i < N; i++)
    {
        hotel[i] = malloc((M / 8)/sizeof(long)+1);
        scanf("%s", line);
        for(j = 0; j < M; j++)
            if(line[j] == '1')
                *(hotel[i]-(j/(sizeof(long)*8))) += 1<<(j%(sizeof(long)*8));
    }
    scanf("%d", &requiered);
    printf("%d", checkPlaces(hotel, requiered, N, M));
    return 0;
}
