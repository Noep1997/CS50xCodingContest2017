#include <stdio.h>
#include <stdlib.h>

typedef struct aabb
{
    float x;
    float y;
    float w;
    float h;
}AABB;

float min(float a, float b)
{
    return a > b ? b : a;
}

float max(float a, float b)
{
    return a < b ? b : a;
}

float xabs(float a)
{
    return a < 0 ? -a : a;
}

void inputAABB(AABB* a)
{
    scanf("%f %f %f %f", &(a->x), &(a->y), &(a->w), &(a->h));
}
float getOverLapping(float x1, float w1, float x2, float w2)
{
    return xabs(min((x1+w1), (x2 + w2)) - max(x1, x2));
}

int main()
{
    AABB a1;
    AABB a2;
    inputAABB(&a1);
    inputAABB(&a2);
    float w = getOverLapping(a1.x, a1.w, a2.x, a2.w);
    float h = getOverLapping(a1.y, a1.h, a2.y, a2.h);
    printf("%.2f", w*h);
    return 0;
}
