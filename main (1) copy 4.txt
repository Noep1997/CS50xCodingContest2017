#include <stdio.h>
#include <stdlib.h>
#define N 100000
int longuestWord(const char* S)
{
    int i, j , longuest = 0;
    for(i = 0; S[i] != '\0'; i++)
    {
        for(j = i; S[j] != '\0' && S[j] != ' '; j++);
        if(longuest < j - i)
            longuest = j - i;
    }
    return longuest;
}
void printSpecialLine(const int l)
{
    int i;
    for(i = 0; i < l; i++)
        printf("#");
}
void printLine(const int l, const char* S)
{
    int i;
    printf("# ");
    for(i = 0; S[i] != '\0'; i++)
        printf("%c", S[i]);
    for(; i < l; i++)
        printf(" ");
    printf(" #\n");
}
void print(const char* S, const int l)
{
    int i, j;
    char n[45] = "";
    printSpecialLine(l+4);
    printf("\n");
    for(i = 0, j = 0; S[i] != '\0'; i++)
    {
        if(S[i] != ' ')
            n[j++] = S[i];
        else
        {
            n[j++] = '\0';
            j = 0;
            printLine(l, n);
        }
    }
    printLine(l, n);
    printSpecialLine(l+4);
}
int main()
{
    char S[N];
    scanf("%[^\n]", S);
    int l = longuestWord(S);
    print(S, l);
    return 0;
}
