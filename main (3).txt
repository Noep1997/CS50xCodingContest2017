
int numerize(const char* S)
{
    int i, sum = 0, tmp = 0;
    for(i = 0; S[i] != '\0'; i++)
    {
        tmp = num(S[i]);
        sum += tmp >= num(S[i + 1]) ? tmp : -tmp;
    }
    return sum;
}
