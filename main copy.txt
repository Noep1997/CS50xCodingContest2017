
int run(char* S)
{

    capitalize(&S[0]);
    for(i = 0; i < N && S[i] != '\0'; i++)
    {
        for(j = i; i < N && S[i] != '\0'&&S[i] != ' '; i++)
            tmp[i-j] = S[i];
        tmp[i - j] = '\0';
        if(checkWord(tmp))
            capitalize(&S[j]);
    }
    capitalize(&S[j]);
    return 0;
}
